class Vecteur:
    def __init__(self, x, y):
        self.x, self.y = x, y
    def __add__(self, other):
        return Vecteur(self.x+other.x, self.y+other.y)
    def __len__(self): return 2
    def __eq__(self, other):
        return self.x==other.x and self.y==other.y
    def __getitem__(self, i):
        return (self.x, self.y)[i]
    def __iter__(self):
        yield self.x; yield self.y

def compteur(n):

def aplatir(lst):
    for item in lst:

Protocole itérateur
class Compte:
    def __init__(self, max):
        self.max = max
        self.n = 0

    def __iter__(self):
        return self

    def __next__(self):
        if self.n >= self.max:
            raise StopIteration
        val = self.n
        self.n += 1
        return val

for x in Compte(3):
    print(x)  # 0 1 2

Décorateur simple
from functools import wraps
import time

def chronomètre(func):
    @wraps(func)  # préserve métadonnées
    def wrapper(*args, **kwargs):
        début = time.time()
        résultat = func(*args, **kwargs)
        durée = time.time() - début
        print(f"{func.__name__} : {durée:.3f}s")
        return résultat
    return wrapper

@chronomètre
def calcul():
    return sum(range(1_000_000))
Décorateur avec paramètres
def répéter(n):
    def décorateur(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for _ in range(n):
                résultat = func(*args, **kwargs)
            return résultat
        return wrapper
    return décorateur

@répéter(3)
def coucou(): print("Bonjour!")

# Métaclasse personnalisée
class Singleton(type):
    _instances = {}
    def __call__(cls, *a, **kw):
        if cls not in cls._instances:
            cls._instances[cls] = super()().__call__(*a, **kw)
        return cls._instances[cls]

class Config(metaclass=Singleton): pass
